JavaScript链式调用原理

发表于 2017-10-22 |

原理

链式调用的原理就是通过构造函数中的每个方法，在结尾都返回实例本身，这样就可以通过调用方法后继续通过实例来进行链式调用：

/* underscore实现方式，与lodash不同，lodash是惰性调用 */
var _ = function (array) {
    // 建立数组副本方法
    this._value = Array.prototype.slice.apply(array)
}

_.forEach = function (array, fn) {
    array.forEach(function (item, index, array) {
        fn.apply(item, [item, index, array])
    })
}
_.map = function (array, fn) {
    return array.map(function (item, index, array) {
        return fn.apply(item, [item, index, array])
    })
}
_.chain = function (array) {
    return new _(array)
}

// 链式调用遍历，实例化后除了chain每个方法都返回实例本身，故可以链式调用
for (var i in _) {
    if (i !== 'chain') {
        // 除去顶端chain，将所有方法赋值给prototype
        _.prototype[i] = (function (i) {
            // 闭包执行全局变量i
            return function (fn) {
                // 取出参数，其实就是fn，相当于以下写法
                // var args = []
                // args.push(fn)
                var args = Array.prototype.slice.apply(arguments)
                // 将value放在数组第一个，用于apply传值
                args.unshift(this._value)
                this._value = _[i].apply(this, args);
                return this;
            }
        })(i);
    }
}

_.prototype.value = function () {
    return this._value
}

var a = [1, 2, 3]

_.forEach(a, function (item) {
    console.log(item)
})

console.log(_.map(a, function (item) {
    return ++item
}))

_.chain(a).map(function (item, index) {
    return item * index
}).value()
